58struct unstoppable_scheduler {
59 using scheduler_concept = typename Sched::scheduler_concept;
60
61 template <typename Q, typename... Args>
62 requires requires { ::std::declval<Sched>().query(::std::declval<const Q&>(), ::std::declval<Args>()...); }
63 auto query(const Q& q, Args&&... args) const noexcept -> decltype(auto) {
64 return sched.query(q, ::std::forward<Args>(args)...);
65 }
66
67 auto schedule() const noexcept(std::is_nothrow_invocable_v<::beman::execution::schedule_t, Sched>) {
68 return ::beman::execution::unstoppable(::beman::execution::schedule(sched));
69 }
70
71 friend auto operator==(const unstoppable_scheduler& lhs, const unstoppable_scheduler& rhs) -> bool = default;

73 Sched sched;
74};
